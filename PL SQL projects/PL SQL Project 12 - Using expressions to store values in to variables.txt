--USING EXPRESSION TO ASSIGN VALUES TO VARIABLES
SET SERVEROUTPUT ON
DECLARE
    V_VAR1 NUMBER;
    V_VAR2 NUMBER;
    V_LOG BOOLEAN;
BEGIN
    V_VAR1 := 100;
    V_VAR2 := 200;
    V_LOG := FALSE;
    IF V_VAR1 >= 50 AND V_VAR2 BETWEEN 1 AND 300
    THEN
        IF NOT(V_LOG) <> TRUE
        THEN
            DBMS_OUTPUT.PUT_LINE('THE VARIABLE IS GREATER THAN 50');
            DBMS_OUTPUT.PUT_LINE('THE VARIABLE 2 IS BETWEEN 1 TO 300');
            V_VAR2 := V_VAR1**0.55;
            DBMS_OUTPUT.PUT_LINE('THE OUTPUT OF THE OPERATION IS: '|| V_VAR2);
        END IF;
        DBMS_OUTPUT.PUT_LINE('THE VALUE OF V_LOG WAS NOT TRUE. HENCE OPERATIONS WEREN''T EXECTUED');
    END IF;
END;
