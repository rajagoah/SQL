SET SERVEROUTPUT ON
DECLARE
    CURSOR C1 IS
        SELECT DEPARTMENT_ID, SUM(SALARY) AS SALARY FROM EMPLOYEE
        WHERE DEPARTMENT_ID IS NOT NULL
        GROUP BY DEPARTMENT_ID
        ORDER BY DEPARTMENT_ID ASC;
    
    V_ROW C1%ROWTYPE;
BEGIN
    DBMS_OUTPUT.PUT_LINE('DEP_ID' || '      '||'SALARY');
    OPEN C1;
    LOOP
        FETCH C1 INTO V_ROW;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(V_ROW.DEPARTMENT_ID||'         '|| V_ROW.SALARY);
    END LOOP;
END;
