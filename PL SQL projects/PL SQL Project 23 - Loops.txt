CREATE OR REPLACE FUNCTION IS_SATURDAY(IN_DATE DATE)
RETURN VARCHAR2
IS
    V_OUT_TX VARCHAR2(222);
    DAY_OF_WEEK_SAT NUMBER := 7;
    DAY_DIFF NUMBER;

BEGIN
--FIND OUT THE DAY OF THE WEEK IT IS
    DAY_DIFF := DAY_OF_WEEK_SAT - TO_CHAR(IN_DATE,'D');
    DBMS_OUTPUT.PUT_LINE(DAY_DIFF);
    DBMS_OUTPUT.PUT_LINE('THE DAY OF THE WEEK THE DATE YOU SUPPLIED IS '|| TO_CHAR(IN_DATE,'D'));
    DBMS_OUTPUT.PUT_LINE('THE DATE OF SATURDAY IS '|| TO_CHAR(IN_DATE + 4,'D'));
--CONDITIONAL STATEMENTS
    CASE 
    WHEN DAY_DIFF = 0 THEN V_OUT_TX := 'THIS IS A SATURDAY';
    WHEN DAY_DIFF <> 0 THEN V_OUT_TX := 'THIS IS NOT A SATURDAY';
    END CASE;
    RETURN V_OUT_TX;
END;
