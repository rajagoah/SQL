-- CREATING A PROCEDURE TO PRINT THE OUTPUT
CREATE OR REPLACE PROCEDURE P_PRINT(NBR IN NUMBER)
IS 
BEGIN
    DBMS_OUTPUT.PUT_LINE(NBR);
END;

-- CREATING A FUNCTION TO CALCULATE THE AREA
CREATE OR REPLACE FUNCTION GET_AREA(LENG NUMBER, BRED NUMBER)
RETURN NUMBER
IS
BEGIN
    RETURN LENG*BRED;
END;

SET SERVEROUTPUT ON
BEGIN
    P_PRINT(GET_AREA(3,5));
END;
