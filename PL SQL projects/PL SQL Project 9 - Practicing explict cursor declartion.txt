--EXAMPLE OF EXPLICIT CURSOR
SET SERVEROUTPUT ON
DECLARE
    CURSOR C1 IS
    SELECT COUNT(*) FROM TB;
    V_COUNT NUMBER;
BEGIN
    OPEN C1;
    FETCH C1 INTO V_COUNT;
    CLOSE C1;
    DBMS_OUTPUT.PUT_LINE('THE COUNT OF RECORDS IS '|| V_COUNT);
END;

Cursors can be used to:
1. Store multiple rows returned by SQL query
2. Loop through and examine multiple rows returned by SQL query
3. Loop through cursor records within another cursor using nesting
4. Change cursor behaviour based on parameter passed


--EXAMPLE OF EXPLICIT CURSOR USING %ROWTYPE DATA TYPE ASSIGNMENT
SET SERVEROUTPUT ON
DECLARE
    CURSOR C1 IS
    SELECT COL1, COL2, COL3 FROM TB;
    V_COL C1%ROWTYPE;
BEGIN
    OPEN C1;
    FETCH C1 INTO V_COL;
    CLOSE C1;
    DBMS_OUTPUT.PUT_LINE(V_COL.COL1||' '||V_COL.COL2||' '||V_COL.COL3);
END;
