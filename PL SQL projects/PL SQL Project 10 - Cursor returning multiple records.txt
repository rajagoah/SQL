--EXAMPLE OF EXPLICIT CURSOR RETURNING MULTIPLE RECORDS
SET SERVEROUTPUT ON
DECLARE
-- DECLARING A CURSOR TO STORE THE VALUES RETURNED BY THE SQL
    CURSOR C1 IS
    SELECT * FROM TB;
    V_COL C1%ROWTYPE;
BEGIN
--SINCE WE HAVE MULTIPLE RECORDS, WE WILL OPEN CURSOR AND THEN LOOP THROUGH THE RECS IN THE CURSOR. THE LOOP WILL EXIT WHEN THERE IS NO MORE RECORDS LEFT IN THE CURSOR
    OPEN C1;
    LOOP    
        FETCH C1 INTO V_COL;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(V_COL.COL1);
    END LOOP;
    CLOSE C1;
END;

--VARIANT OF ABOVE CODE
SET SERVEROUTPUT ON
DECLARE
    CURSOR C1 IS
    SELECT * FROM TB;
    V_COL C1%ROWTYPE;
BEGIN
    OPEN C1;
    LOOP    
        FETCH C1 INTO V_COL;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(V_COL.COL1||' '||V_COL.COL2||' '||V_COL.COL3);
    END LOOP;
    CLOSE C1;
END;
