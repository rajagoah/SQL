SET SERVEROUTPUT ON
CREATE OR REPLACE FUNCTION INCOME_LEVEL (EMP_ID IN VARCHAR2)
RETURN VARCHAR2
IS
	INCOME INTEGER;
	INCOME_LEVEL VARCHAR(20);

--CREATING A CURSOR
	CURSOR C1 IS
		SELECT SALARY FROM EMPLOYEE
		WHERE EMPLOYEE_ID = EMP_ID;
BEGIN
	OPEN C1;
	FETCH C1 INTO INCOME;
	CLOSE C1;

IF INCOME < 50000 THEN
	INCOME_LEVEL := 'LOW INCOME';
ELSIF INCOME >= 50000 AND INCOME < 100000 THEN
	INCOME_LEVEL := 'MID INCOME';
ELSE INCOME_LEVEL := 'HIGH INCOME';
END IF;
RETURN INCOME_LEVEL;
END;

set serveroutput on
BEGIN
DBMS_OUTPUT.PUT_LINE(INCOME_LEVEL(28));
END;
