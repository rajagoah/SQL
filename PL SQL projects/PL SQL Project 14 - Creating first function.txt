CREATE OR REPLACE FUNCTION F_GET_AREA (V_PI IN NUMBER, I_RAD1 IN NUMBER := 1)
RETURN NUMBER
IS
BEGIN
    RETURN V_PI*(I_RAD1*I_RAD1);
END;

SET SERVEROUTPUT ON
DECLARE
    V_PI NUMBER := 3.14;
BEGIN
    DBMS_OUTPUT.PUT_LINE(F_GET_AREA(V_PI));
    DBMS_OUTPUT.PUT_LINE(F_GET_AREA(V_PI, 3));
    DBMS_OUTPUT.PUT_LINE(F_GET_AREA(V_PI, 4));
END;
