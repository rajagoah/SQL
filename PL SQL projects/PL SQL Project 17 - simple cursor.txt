    SET SERVEROUTPUT ON
    DECLARE
        CURSOR C1 IS
        SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME, SALARY FROM EMPLOYEE WHERE DEPARTMENT_ID = 10;
        V_ROW C1%ROWTYPE;
    BEGIN
        OPEN C1;
        LOOP
            FETCH C1 INTO V_ROW;
            EXIT WHEN C1%NOTFOUND;
                DBMS_OUTPUT.PUT_LINE(V_ROW.EMPLOYEE_ID||'       '||V_ROW.FIRST_NAME||'      '||V_ROW.LAST_NAME||'   '||V_ROW.SALARY);
        END LOOP;
        CLOSE C1;
    END;
