CREATE OR REPLACE FUNCTION F_GET_DIFF ( I_RAD1 NUMBER, I_RAD2 NUMBER)
RETURN NUMBER
IS
    V_AREA1 NUMBER;
    V_AREA2 NUMBER;
    V_OUT   NUMBER;
    
    FUNCTION F_GET_AREA(I_RAD NUMBER)
    RETURN NUMBER
    IS
    BEGIN
        RETURN 3.14 * I_RAD**2;
    END;
BEGIN
    V_AREA1 := F_GET_AREA(I_RAD1);
    V_AREA2 := F_GET_AREA(I_RAD2);
    V_OUT := V_AREA1 - V_AREA2;
    RETURN V_OUT;
END;

SET SERVEROUTPUT ON
BEGIN
    DBMS_OUTPUT.PUT_LINE(F_GET_DIFF(5,3));
    DBMS_OUTPUT.PUT_LINE(F_GET_DIFF(5,6));
END;
