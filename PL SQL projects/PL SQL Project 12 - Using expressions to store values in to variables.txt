--USING EXPRESSION TO ASSIGN VALUES TO VARIABLES
SET SERVEROUTPUT ON
DECLARE
    V_VAR1 NUMBER;
    V_VAR2 NUMBER;
BEGIN
    V_VAR1 := 100;
    IF V_VAR1 >= 50
    THEN
        DBMS_OUTPUT.PUT_LINE('THE VARIABLE IS GREATER THAN 50');
        V_VAR2 := V_VAR1**0.55;
        DBMS_OUTPUT.PUT_LINE('THE OUTPUT OF THE OPERATION IS: '|| V_VAR2);
    END IF;
END;
