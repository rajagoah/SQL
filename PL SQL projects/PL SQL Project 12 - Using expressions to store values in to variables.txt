--USING EXPRESSION TO ASSIGN VALUES TO VARIABLES
SET SERVEROUTPUT ON
DECLARE
    V_LEN NUMBER;
    V_WID NUMBER;
    V_HEI NUMBER;
    V_VOL NUMBER;
    V_LOG BOOLEAN;
BEGIN
    V_LEN := 100;
    V_WID := 200;
    V_HEI := 150;
    V_LOG := FALSE;

    IF NOT(V_LOG) = TRUE
    THEN
        V_VOL := V_LEN * V_WID * V_HEI;
        DBMS_OUTPUT.PUT_LINE('THE VOLUME OF THE CUBE IS: '|| V_VOL);
    ELSE
        DBMS_OUTPUT.PUT_LINE('THE VALUE OF V_LOG WAS NOT TRUE. HENCE OPERATIONS WEREN''T EXECTUED');
    END IF;
END;
