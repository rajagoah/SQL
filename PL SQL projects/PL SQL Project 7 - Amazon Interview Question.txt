--Command to make the PL/SQL to the console
SET SERVEROUTPUT ON

--creating a function. This function accepts an argument
CREATE OR REPLACE FUNCTION AMAZON_INTERVIEW_2 (ST IN VARCHAR2)
RETURN VARCHAR2
IS

    --declaring a variable of data type of value in the states table
    V_STATES_2 STATES.STATES%TYPE;

--begin block to execute a SQL statement
BEGIN
    SELECT STATES_2 INTO V_STATES_2 FROM STATES
    WHERE STATES = ST;
    RETURN 'THE VALUE IN THE SECOND COLUMNS IS --> '|| V_STATES_2;
END;


--This PLSQL command is being used to call on the function
SET SERVEROUTPUT ON
BEGIN
    DBMS_OUTPUT.PUT_LINE(AMAZON_INTERVIEW_2('NY'));
END;
