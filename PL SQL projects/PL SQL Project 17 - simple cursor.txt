SET SERVEROUTPUT ON
DECLARE
    CURSOR C1 IS
    SELECT COL1, COL2, COL3 FROM TB;
    V_ROW C1%ROWTYPE;
BEGIN
    OPEN C1;
    LOOP
        FETCH C1 INTO V_ROW;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(V_ROW.COL1||' '||V_ROW.COL2||' '||V_ROW.COL3);
    END LOOP;
    CLOSE C1;
END;
