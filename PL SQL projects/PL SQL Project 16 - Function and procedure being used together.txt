--PROCEUDRE TO PRINT OUTPUT TO THE CONSOLE
CREATE OR REPLACE PROCEDURE P_PRINT(INPT IN NUMBER)
IS
BEGIN
    DBMS_OUTPUT.PUT_LINE(INPT);
END;

--FUNCTION TO CALCULATE THE DIFFERENCE IN THE AREAS OF 2 RECTANGLES
CREATE OR REPLACE FUNCTION GET_DIFF (RAD1 NUMBER, RAD2 NUMBER)
RETURN NUMBER IS
    V_AREA1 NUMBER;
    V_AREA2 NUMBER;
    V_DIFFR NUMBER;  
    FUNCTION GET_AREA(RAD NUMBER, PI NUMBER := 3.14)
    RETURN NUMBER IS
    BEGIN
        RETURN PI * RAD**2;
    END;
BEGIN
    V_AREA1 := GET_AREA(RAD1);
    V_AREA2 := GET_AREA(RAD2);
    V_DIFFR := V_AREA1 - V_AREA2;
    P_PRINT(V_DIFFR);
    RETURN V_DIFFR;
END;
